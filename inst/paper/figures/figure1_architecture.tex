    },
    arrow/.style={
        ->,
        >=Stealth,
        thick,
        black!70
    },
    bidiarrow/.style={

    },
    label/.style={
        font=\footnotesize\bfseries,
        text=black!80
    },
    sublabel/.style={
        font=\scriptsize,
        text=black!60
